cmake_minimum_required(VERSION 3.1)
project(yolo11_ros)

# ========== 关键修改1：注释原OpenCV查找，手动指定4.5路径 ==========
# find_package(OpenCV REQUIRED)
find_package(OpenCV 4.5 REQUIRED)
# 1. 指定OpenCV 4.5头文件目录
include_directories(/usr/local/opencv45/include/opencv4)
# 2. 指定OpenCV 4.5库目录
link_directories(/usr/local/opencv45/lib)
# 3. 手动定义OpenCV 4.5的库列表（覆盖原OpenCV_LIBS）
set(OpenCV_LIBS 
    /usr/local/opencv45/lib/libopencv_core.so
    /usr/local/opencv45/lib/libopencv_imgproc.so
    /usr/local/opencv45/lib/libopencv_imgcodecs.so
    /usr/local/opencv45/lib/libopencv_dnn.so
    /usr/local/opencv45/lib/libopencv_highgui.so
    /usr/local/opencv45/lib/libopencv_videoio.so
)

# 配置 TensorRT 路径
set(TENSORRT_ROOT /opt/TensorRT-8.5.1.7)
include_directories(${TENSORRT_ROOT}/include)
link_directories(${TENSORRT_ROOT}/lib)

# 配置 CUDA（适配低版本 CMake + Jetson aarch64架构）
set(CMAKE_CUDA_ARCHITECTURES 53 62 72 87)  # 适配Jetson Nano/Xavier/NX
set(CMAKE_CUDA_COMPILER /usr/local/cuda-11.4/bin/nvcc)
find_package(CUDA 11.4 REQUIRED)  # 使用旧版 CUDA 查找方式

# 编译选项
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++14 -O3 -Wl,-rpath=/usr/local/opencv45/lib")
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_BUILD_TYPE Release)
option(CUDA_USE_STATIC_CUDA_RUNTIME OFF)

# ========== 关键修改2：修正CUDA库路径（Jetson是aarch64，不是x86_64） ==========
# 路径配置
set(CUDA_INCLUDE_DIRS ${CUDA_INCLUDE_DIRS} /usr/local/cuda-11.4/include)
set(CUDA_LIBRARIES_DIR /usr/local/cuda-11.4/targets/aarch64-linux/lib)  # 修正为aarch64
set(TensorRT_INCLUDE_DIRS ${TENSORRT_ROOT}/include)
set(TensorRT_LIBRARIES ${TENSORRT_ROOT}/lib)

# 打印路径信息（用于调试）
message(STATUS "CUDA Headers: \n${CUDA_INCLUDE_DIRS}\n")
message(STATUS "CUDA Libs Dir: \n${CUDA_LIBRARIES_DIR}\n")
message(STATUS "TensorRT Headers: \n${TensorRT_INCLUDE_DIRS}\n")
message(STATUS "TensorRT Libs: \n${TensorRT_LIBRARIES}\n")
# 新增：打印OpenCV 4.5路径，确认生效
message(STATUS "OpenCV 4.5 Libs: \n${OpenCV_LIBS}\n")

# 寻找其他依赖包（保留）
find_package(PCL REQUIRED)
find_package(catkin REQUIRED COMPONENTS
  roscpp
  roslib
  rospy
  cv_bridge
  sensor_msgs
)

# Catkin 配置
catkin_package()

# 包含头文件路径
include_directories(
  include
  ${PCL_INCLUDE_DIRS}
  ${CUDA_INCLUDE_DIRS}
  ${TensorRT_INCLUDE_DIRS}
  ${catkin_INCLUDE_DIRS}
)

# 链接库路径
link_directories(
  ${PCL_LIBRARY_DIRS}
  ${CUDA_LIBRARIES_DIR}
  ${TensorRT_LIBRARIES}
)

add_compile_options(-Wno-deprecated-declarations)
add_definitions(${PCL_DEFINITIONS})

# ------------------------------
# 定义目标并链接库（保留，OpenCV_LIBS已强制指向4.5）
# ------------------------------

# 1. detect 节点
add_executable(detect src_ros/detect.cpp)
target_link_libraries(detect 
  ${catkin_LIBRARIES}
  ${OpenCV_LIBS}
  ${PCL_LIBRARIES}
  ${CUDA_LIBRARIES}  # 链接CUDA库
  nvinfer
  nvinfer_plugin
)

# 2. pose 节点
add_executable(pose src_ros/pose.cpp)
target_link_libraries(pose 
  ${catkin_LIBRARIES}
  ${OpenCV_LIBS}
  ${PCL_LIBRARIES}
  ${CUDA_LIBRARIES}  # 链接CUDA库
  nvinfer
  nvinfer_plugin
)

# 3. segment 节点
add_executable(segment src_ros/segment.cpp)
target_link_libraries(segment 
  ${catkin_LIBRARIES}
  ${OpenCV_LIBS}
  ${PCL_LIBRARIES}
  ${CUDA_LIBRARIES}  # 链接CUDA库
  nvinfer
  nvinfer_plugin
)

install(DIRECTORY launch/
  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}/launch
  FILES_MATCHING PATTERN "*.launch"  # 匹配所有 .launch 文件
)
